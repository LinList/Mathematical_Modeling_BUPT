## 一、评价类模型

### 1.1 AHP 层次分析法

AHP（Analytic Hierarchy Process）用于多准则决策。核心是通过两两比较确定各指标的权重。

```python
import numpy as np

# ========================================
# AHP 层次分析法完整实现
# ========================================

def ahp(criteria_names, comparison_matrix):
    """
    AHP层次分析法

    参数:
        criteria_names: 指标名称列表，如 ['价格', '质量', '服务']
        comparison_matrix: 判断矩阵（两两比较结果）
            取值含义: 1=同等重要, 3=稍重要, 5=明显重要, 7=强烈重要, 9=极端重要
            2,4,6,8 为中间值，倒数表示反向比较

    返回:
        weights: 各指标权重
        CR: 一致性比率（CR<0.1 则通过一致性检验）
    """
    n = len(criteria_names)
    matrix = np.array(comparison_matrix, dtype=float)

    # ====== 第一步：计算权重（特征向量法） ======
    # 方法：求矩阵的最大特征值和对应特征向量

    # 计算每列的和
    col_sums = matrix.sum(axis=0)

    # 归一化：每个元素除以列和
    normalized = matrix / col_sums

    # 每行求均值，得到权重向量
    weights = normalized.mean(axis=1)

    print("=== AHP 层次分析法 ===")
    print(f"\n判断矩阵:\n{matrix}")
    print(f"\n归一化矩阵:\n{normalized}")
    print(f"\n权重向量:")
    for name, w in zip(criteria_names, weights):
        print(f"  {name}: {w:.4f}")

    # ====== 第二步：一致性检验 ======
    # 计算最大特征值 lambda_max
    # lambda_max = (1/n) * sum(W_i / w_i)，其中 W = A * w
    Aw = matrix @ weights
    lambda_max = (Aw / weights).mean()

    # 一致性指标 CI = (lambda_max - n) / (n - 1)
    CI = (lambda_max - n) / (n - 1)

    # 随机一致性指标 RI（查表获得，n 为矩阵阶数）
    RI_table = {1: 0, 2: 0, 3: 0.58, 4: 0.90, 5: 1.12,
                6: 1.24, 7: 1.32, 8: 1.41, 9: 1.45, 10: 1.49}
    RI = RI_table.get(n, 1.49)

    # 一致性比率 CR = CI / RI
    CR = CI / RI if RI > 0 else 0

    print(f"\n最大特征值 lambda_max = {lambda_max:.4f}")
    print(f"一致性指标 CI = {CI:.4f}")
    print(f"随机一致性指标 RI = {RI}")
    print(f"一致性比率 CR = {CR:.4f}")

    if CR < 0.1:
        print(f"\n✓ 一致性检验通过！（CR = {CR:.4f} < 0.1）")
    else:
        print(f"\n✗ 一致性检验未通过！（CR = {CR:.4f} >= 0.1）")
        print("  请调整判断矩阵，使其更具一致性")

    return weights, CR


# ====== 示例：选择最佳供应商 ======
# 评价标准：价格、质量、交货期、服务
criteria = ['价格', '质量', '交货期', '服务']

# 判断矩阵（4x4）
# 含义：第i行第j列表示"指标i相对指标j的重要性"
# 例如 matrix[0][1] = 3 表示"价格"比"质量"稍微重要
comparison_matrix = [
    [1,     3,   5,   7],   # 价格 vs 其他
    [1/3,   1,   3,   5],   # 质量 vs 其他
    [1/5, 1/3,   1,   3],   # 交货期 vs 其他
    [1/7, 1/5, 1/3,   1]    # 服务 vs 其他
]

weights, CR = ahp(criteria, comparison_matrix)

# ====== 计算各方案的综合得分 ======
print("\n=== 各供应商综合评分 ===")

# 三个供应商在各指标上的得分（1-10分）
suppliers = ['供应商A', '供应商B', '供应商C']
scores = np.array([
    [8, 9, 7, 8],   # 供应商A：价格8, 质量9, 交货期7, 服务8
    [9, 7, 8, 7],   # 供应商B
    [7, 8, 9, 9]    # 供应商C
])

# 加权得分 = 各指标得分 × 权重
weighted_scores = scores @ weights

for name, score in zip(suppliers, weighted_scores):
    print(f"  {name}: {score:.4f}")

best = suppliers[np.argmax(weighted_scores)]
print(f"\n推荐选择: {best}（得分最高）")
```

> **避坑指南**：判断矩阵必须满足 `a[i][j] = 1/a[j][i]`，即正反两个方向的比较值互为倒数。如果 CR >= 0.1，需要调整判断矩阵。常见的调整方法是让比较更"均匀"。

### 1.2 TOPSIS 法

TOPSIS（逼近理想解排序法）通过计算每个方案与"最优解"和"最劣解"的距离来排序。

```python
import numpy as np

# ========================================
# TOPSIS 逼近理想解排序法完整实现
# ========================================

def topsis(decision_matrix, weights, criteria_types):
    """
    TOPSIS法

    参数:
        decision_matrix: 决策矩阵（方案数 × 指标数）
        weights: 各指标权重
        criteria_types: 各指标类型列表
            'benefit' 表示效益型（越大越好）
            'cost' 表示成本型（越小越好）

    返回:
        scores: 各方案的相对接近度（0~1，越大越好）
        ranking: 排序结果
    """
    matrix = np.array(decision_matrix, dtype=float)
    weights = np.array(weights, dtype=float)
    n_schemes, n_criteria = matrix.shape

    print("=== TOPSIS 逼近理想解排序法 ===")
    print(f"\n原始决策矩阵:\n{matrix}")

    # ====== 第一步：正向化 ======
    # 将所有指标转化为"越大越好"的效益型指标
    normalized_matrix = matrix.copy()
    for j in range(n_criteria):
        if criteria_types[j] == 'cost':
            # 成本型指标：取倒数 或 用最大值减去
            # 方法一：取倒数
            # normalized_matrix[:, j] = 1.0 / matrix[:, j]
            # 方法二：用最大值减去
            normalized_matrix[:, j] = matrix[:, j].max() - matrix[:, j]

    print(f"\n正向化后的矩阵:\n{normalized_matrix}")

    # ====== 第二步：标准化 ======
    # 每个元素除以所在列的平方和的平方根
    col_norms = np.sqrt((normalized_matrix ** 2).sum(axis=0))
    standard_matrix = normalized_matrix / col_norms

    print(f"\n标准化后的矩阵:\n{standard_matrix}")

    # ====== 第三步：加权标准化 ======
    weighted_matrix = standard_matrix * weights

    print(f"\n加权标准化矩阵:\n{weighted_matrix}")

    # ====== 第四步：确定正理想解和负理想解 ======
    # 正理想解：每列最大值
    # 负理想解：每列最小值
    positive_ideal = weighted_matrix.max(axis=0)
    negative_ideal = weighted_matrix.min(axis=0)

    print(f"\n正理想解: {positive_ideal}")
    print(f"负理想解: {negative_ideal}")

    # ====== 第五步：计算各方案到理想解的距离 ======
    # 欧氏距离
    d_positive = np.sqrt(((weighted_matrix - positive_ideal) ** 2).sum(axis=1))
    d_negative = np.sqrt(((weighted_matrix - negative_ideal) ** 2).sum(axis=1))

    print(f"\n到正理想解的距离: {d_positive}")
    print(f"到负理想解的距离: {d_negative}")

    # ====== 第六步：计算相对接近度 ======
    # Ci = d_negative / (d_positive + d_negative)
    # Ci 越大，方案越优
    scores = d_negative / (d_positive + d_negative)

    print(f"\n相对接近度: {scores}")

    # ====== 排序 ======
    ranking = np.argsort(scores)[::-1]  # 降序排列

    return scores, ranking


# ====== 示例：评价5个城市的综合发展水平 ======
# 指标：GDP(亿元)、人均收入(万元)、绿化率(%)、PM2.5(μg/m³)
# 权重由AHP确定（这里直接给定）
# PM2.5是成本型指标（越小越好），其余为效益型

cities = ['城市A', '城市B', '城市C', '城市D', '城市E']

# 决策矩阵（5个城市 × 4个指标）
decision_matrix = [
    [8500, 5.2, 42, 65],   # 城市A
    [7200, 4.8, 38, 55],   # 城市B
    [9100, 5.5, 45, 75],   # 城市C
    [6800, 4.5, 35, 48],   # 城市D
    [8800, 5.0, 40, 60]    # 城市E
]

# 各指标权重（权重之和为1）
weights = [0.3, 0.25, 0.2, 0.25]

# 指标类型
criteria_types = ['benefit', 'benefit', 'benefit', 'cost']

scores, ranking = topsis(decision_matrix, weights, criteria_types)

print("\n=== 最终排序结果 ===")
for rank, idx in enumerate(ranking):
    print(f"  第{rank+1}名: {cities[idx]}（接近度 = {scores[idx]:.4f}）")
```

> **小贴士**：TOPSIS 的关键在于正确区分"效益型"和"成本型"指标。如果搞反了，结果就完全错了。正向化时，成本型指标可以用取倒数或最大值减去两种方法。

### 1.3 熵权法

熵权法是一种客观赋权方法，根据数据的离散程度确定权重。数据差异越大，权重越高。

```python
import numpy as np

# ========================================
# 熵权法完整实现
# ========================================

def entropy_weight(decision_matrix, criteria_types=None):
    """
    熵权法（客观赋权）

    参数:
        decision_matrix: 决策矩阵（方案数 × 指标数）
        criteria_types: 指标类型列表，'benefit'或'cost'（可选，用于正向化）

    返回:
        weights: 各指标的客观权重
    """
    matrix = np.array(decision_matrix, dtype=float)
    n_schemes, n_criteria = matrix.shape

    print("=== 熵权法 ===")
    print(f"\n原始决策矩阵:\n{matrix}")

    # ====== 第一步：正向化（可选） ======
    if criteria_types is not None:
        for j in range(n_criteria):
            if criteria_types[j] == 'cost':
                matrix[:, j] = matrix[:, j].max() - matrix[:, j]
        print(f"\n正向化后:\n{matrix}")

    # ====== 第二步：标准化 ======
    # 每个元素除以列和，使每列之和为1
    col_sums = matrix.sum(axis=0)
    normalized = matrix / col_sums

    print(f"\n标准化后:\n{normalized}")

    # ====== 第三步：计算信息熵 ======
    # E_j = -k * sum(p_ij * ln(p_ij))，其中 k = 1/ln(n)
    # 注意：p_ij=0 时，p_ij*ln(p_ij) = 0

    k = 1.0 / np.log(n_schemes)
    entropy = np.zeros(n_criteria)

    for j in range(n_criteria):
        p = normalized[:, j]
        # 避免log(0)的计算
        p_nonzero = p[p > 0]
        entropy[j] = -k * np.sum(p_nonzero * np.log(p_nonzero))

    print(f"\n信息熵: {entropy}")

    # ====== 第四步：计算权重 ======
    # 信息效用值 d_j = 1 - E_j
    # 权重 w_j = d_j / sum(d_j)
    d = 1 - entropy
    weights = d / d.sum()

    print(f"\n信息效用值: {d}")
    print(f"\n熵权法权重:")
    for i, w in enumerate(weights):
        print(f"  指标{i+1}: {w:.4f}")

    return weights


# ====== 示例：评价5个方案 ======
# 4个指标，数据差异越大说明区分度越高，权重也越高
decision_matrix = [
    [10, 50, 0.5, 100],
    [20, 60, 0.6, 80],
    [30, 55, 0.4, 90],
    [15, 70, 0.8, 70],
    [25, 65, 0.7, 85]
]

weights = entropy_weight(decision_matrix)

# ====== 结合TOPSIS使用 ======
print("\n=== 熵权法 + TOPSIS 综合评价 ===")
# 用熵权法得到的权重，跑TOPSIS
scores, ranking = topsis(decision_matrix, weights,
                          ['benefit', 'benefit', 'benefit', 'cost'])
```

> **小贴士**：熵权法和AHP可以结合使用——AHP反映主观经验，熵权法反映客观数据。常用做法：最终权重 = 0.5 × AHP权重 + 0.5 × 熵权法权重。

### 1.4 模糊综合评价

模糊综合评价用于处理"边界不清晰"的评价问题，如"服务质量好不好"这种难以精确量化的指标。

```python
import numpy as np

# ========================================
# 模糊综合评价完整实现
# ========================================

def fuzzy_evaluation(weight_matrix, evaluation_matrix):
    """
    模糊综合评价

    参数:
        weight_matrix: 各因素的权重向量，如 [0.3, 0.3, 0.2, 0.2]
        evaluation_matrix: 模糊关系矩阵（因素数 × 评语数）
            每行表示某因素对各评语等级的隶属度，每行之和为1

    返回:
        result: 综合评价结果向量
        grade: 最终评价等级
    """
    weights = np.array(weight_matrix, dtype=float)
    R = np.array(evaluation_matrix, dtype=float)

    print("=== 模糊综合评价 ===")
    print(f"\n权重向量: {weights}")
    print(f"\n模糊关系矩阵 R:\n{R}")

    # ====== 模糊合成运算 ======
    # 方法一：普通矩阵乘法（加权平均型，最常用）
    result = weights @ R

    # 方法二：取小取大运算（主因素决定型）
    # result_min_max = np.zeros(R.shape[1])
    # for j in range(R.shape[1]):
    #     result_min_max[j] = np.min(weights * R[:, j])  # 不对，应该先取小再取大
    # 正确的取小取大：
    # for j in range(R.shape[1]):
    #     result_min_max[j] = np.max(np.minimum(weights, R[:, j]))

    # 归一化（确保和为1）
    result = result / result.sum()

    print(f"\n综合评价结果: {result}")

    # ====== 确定评价等级 ======
    # 取最大隶属度对应的等级
    grades = ['很差', '较差', '一般', '较好', '很好']
    best_idx = np.argmax(result)

    print(f"\n各等级隶属度:")
    for grade, score in zip(grades, result):
        bar = '█' * int(score * 50)
        print(f"  {grade}: {score:.4f} {bar}")

    print(f"\n最终评价: {grades[best_idx]}（隶属度 = {result[best_idx]:.4f}）")

    return result, grades[best_idx]


# ====== 示例：评价某餐厅的综合服务质量 ======
# 评价因素：口味、价格、环境、服务
factors = ['口味', '价格', '环境', '服务']

# 各因素权重（通过AHP确定）
weights = [0.35, 0.25, 0.20, 0.20]

# 模糊关系矩阵 R（4个因素 × 5个评语等级）
# 评语等级：很差、较差、一般、较好、很好
# 每行是某因素对各等级的隶属度（通过问卷调查统计得到）
R = [
    [0.00, 0.05, 0.15, 0.50, 0.30],  # 口味：50%较好，30%很好
    [0.05, 0.10, 0.40, 0.35, 0.10],  # 价格：40%一般，35%较好
    [0.00, 0.00, 0.20, 0.45, 0.35],  # 环境：45%较好，35%很好
    [0.00, 0.05, 0.25, 0.50, 0.20]   # 服务：50%较好，20%很好
]

result, grade = fuzzy_evaluation(weights, R)

# ====== 加权得分法（补充） ======
print("\n=== 加权得分法 ===")
# 给各等级赋分：很差=1, 较差=2, 一般=3, 较好=4, 很好=5
scores = np.array([1, 2, 3, 4, 5])
final_score = result @ scores
print(f"综合得分: {final_score:.2f} / 5.00")
```

## 二、优化类模型

### 2.1 线性规划

```python
import numpy as np
from scipy.optimize import linprog

# ========================================
# 线性规划（scipy.optimize.linprog）
# ========================================

# ====== 示例：生产计划问题 ======
# 某工厂生产两种产品A和B
# 产品A：利润2万元/件，需原料甲2吨、原料乙1吨
# 产品B：利润3万元/件，需原料甲1吨、原料乙2吨
# 原料甲总供应量：8吨，原料乙总供应量：8吨
# 求：生产多少A和B，使利润最大？

# scipy.linprog 求解的是最小化问题，所以要取负号转化为最小化
# min -2x1 - 3x2
# s.t. 2x1 + x2 <= 8   （原料甲约束）
#      x1 + 2x2 <= 8   （原料乙约束）
#      x1, x2 >= 0

print("=== 线性规划：生产计划问题 ===")

# 目标函数系数（注意取负号，因为linprog求最小值）
c = [-2, -3]

# 不等式约束矩阵（Ax <= b）
A_ub = [
    [2, 1],   # 原料甲：2x1 + x2 <= 8
    [1, 2]    # 原料乙：x1 + 2x2 <= 8
]
b_ub = [8, 8]

# 变量下界（x1, x2 >= 0）
bounds = [(0, None), (0, None)]

# 求解
result = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

print(f"\n求解状态: {result.message}")
print(f"\n最优解:")
print(f"  产品A生产: {result.x[0]:.2f} 件")
print(f"  产品B生产: {result.x[1]:.2f} 件")
print(f"  最大利润: {-result.fun:.2f} 万元")

# ====== 带等式约束的例子 ======
print("\n=== 线性规划：带等式约束 ===")

# min 3x1 + 5x2
# s.t. x1 + x2 = 10      （等式约束）
#      x1 >= 3            （下界约束）
#      x2 >= 2            （下界约束）

c2 = [3, 5]
A_eq = [[1, 1]]   # 等式约束矩阵
b_eq = [10]        # 等式约束右端项
bounds2 = [(3, None), (2, None)]

result2 = linprog(c2, A_eq=A_eq, b_eq=b_eq, bounds=bounds2, method='highs')

print(f"最优解: x1={result2.x[0]:.2f}, x2={result2.x[1]:.2f}")
print(f"最小值: {result2.fun:.2f}")
```

> **避坑指南**：`scipy.optimize.linprog` 求解的是**最小化**问题。如果你的目标是最大化，需要将目标函数系数取负号。约束条件必须是 `<=` 形式，如果是 `>=` 要乘以 -1 转化。

### 2.2 整数规划

```python
# ========================================
# 整数规划（使用 pulp 库）
# ========================================
# 安装: pip install pulp

from pulp import *

print("=== 整数规划：背包问题 ===")

# 问题：背包容量为10kg，有4个物品，求在容量范围内价值最大
items = ['物品1', '物品2', '物品3', '物品4']
weights = [3, 4, 5, 2]   # 重量（kg）
values = [4, 5, 6, 3]    # 价值

# 创建问题实例
prob = LpProblem("背包问题", LpMaximize)  # 最大化

# 创建决策变量（0-1变量：选或不选）
x = [LpVariable(f'x{i}', cat='Binary') for i in range(len(items))]

# 目标函数：最大化总价值
prob += lpSum(values[i] * x[i] for i in range(len(items)))

# 约束条件：总重量不超过容量
prob += lpSum(weights[i] * x[i] for i in range(len(items))) <= 10

# 求解
prob.solve(PULP_CBC_CMD(msg=0))

print(f"求解状态: {LpStatus[prob.status]}")
print(f"\n最优方案:")
total_weight = 0
total_value = 0
for i in range(len(items)):
    if x[i].varValue > 0:
        print(f"  选择 {items[i]}（重量{weights[i]}kg, 价值{values[i]}）")
        total_weight += weights[i]
        total_value += values[i]

print(f"\n总重量: {total_weight}kg / 10kg")
print(f"总价值: {total_value}")

# ====== 混合整数规划示例 ======
print("\n=== 混合整数规划：工厂选址 ===")

# 某公司要在3个候选地址中选建工厂
# 每个工厂有固定建设成本和产能，需满足总需求
sites = ['地址A', '地址B', '地址C']
fixed_costs = [100, 120, 80]   # 固定建设成本（万元）
capacities = [80, 70, 50]      # 产能
total_demand = 100              # 总需求

prob2 = LpProblem("工厂选址", LpMinimize)

# 决策变量：是否建厂（0-1变量）
y = [LpVariable(f'y{i}', cat='Binary') for i in range(3)]

# 决策变量：各工厂产量（连续变量，>=0）
q = [LpVariable(f'q{i}', lowBound=0) for i in range(3)]

# 目标函数：最小化总成本（固定成本 + 产量成本）
# 这里简化为只考虑固定成本
prob2 += lpSum(fixed_costs[i] * y[i] for i in range(3))

# 约束1：总产量 >= 总需求
prob2 += lpSum(q[i] for i in range(3)) >= total_demand

# 约束2：各工厂产量不超过其产能（且只有建厂才能生产）
for i in range(3):
    prob2 += q[i] <= capacities[i] * y[i]

prob2.solve(PULP_CBC_CMD(msg=0))

print(f"求解状态: {LpStatus[prob2.status]}")
print(f"\n最优方案:")
for i in range(3):
    if y[i].varValue > 0:
        print(f"  在{sites[i]}建厂，产量: {q[i].varValue:.1f}")
    else:
        print(f"  不在{sites[i]}建厂")
print(f"总成本: {value(prob2.objective):.1f} 万元")
```

> **小贴士**：`pulp` 库比 `scipy` 更适合做整数规划，语法更直观。`LpMaximize` 表示最大化，`LpMinimize` 表示最小化。变量类型：`'Binary'` 是0-1变量，`'Integer'` 是整数变量，不指定则默认连续变量。

### 3.1 灰色预测 GM(1,1)

灰色预测适用于"少数据、贫信息"的预测问题，通常只需要4个以上的数据点。

```python
import numpy as np

# ========================================
# 灰色预测 GM(1,1) —— 纯 Python 实现
# ========================================

def gm11(x0, predict_steps=3):
    """
    灰色预测 GM(1,1)

    参数:
        x0: 原始数据序列（一维数组）
        predict_steps: 预测步数

    返回:
        predict_values: 预测值列表
        a: 发展系数
        b: 灰作用量
        relative_errors: 相对误差
    """
    x0 = np.array(x0, dtype=float)
    n = len(x0)

    print("=== 灰色预测 GM(1,1) ===")
    print(f"原始数据: {x0}")

    # ====== 第一步：累加生成（AGO） ======
    # x1 是 x0 的前缀和（一次累加序列）
    x1 = np.cumsum(x0)
    print(f"累加序列 x1: {x1}")

    # ====== 第二步：构造矩阵 B 和向量 Y ======
    # B = [[-0.5(x1[1]+x1[0]), 1],
    #      [-0.5(x1[2]+x1[1]), 1],
    #      ...                           ]
    # Y = [x0[1], x0[2], ...]^T

    B = np.zeros((n - 1, 2))
    for i in range(n - 1):
        B[i, 0] = -0.5 * (x1[i + 1] + x1[i])
        B[i, 1] = 1

    Y = x0[1:].reshape(-1, 1)

    # ====== 第三步：最小二乘法求参数 a 和 b ======
    # [a, b]^T = (B^T B)^(-1) B^T Y
    BTB = B.T @ B
    BTB_inv = np.linalg.inv(BTB)
    BTY = B.T @ Y
    params = BTB_inv @ BTY

    a = params[0, 0]  # 发展系数
    b = params[1, 0]  # 灰作用量

    print(f"\n发展系数 a = {a:.6f}")
    print(f"灰作用量 b = {b:.6f}")

    # 检查 a 的范围
    if abs(a) < 0.3:
        print("a值适中，预测效果较好")
    elif abs(a) < 0.5:
        print("⚠ a值偏大，短期预测尚可")
    else:
        print("⚠ a值过大，预测效果可能较差")

    # ====== 第四步：建立预测模型 ======
    # x1 的预测公式: x1_hat(k+1) = (x0[0] - b/a) * exp(-a*k) + b/a

    def predict_x1(k):
        """预测第k步的累加值"""
        return (x0[0] - b / a) * np.exp(-a * k) + b / a

    # ====== 第五步：还原为原始数据的预测值 ======
    # x0_hat(k) = x1_hat(k) - x1_hat(k-1)

    # 拟合原始数据
    x0_hat = np.zeros(n)
    x0_hat[0] = x0[0]  # 第一个值不变
    for k in range(1, n):
        x0_hat[k] = predict_x1(k) - predict_x1(k - 1)

    print(f"\n拟合值: {x0_hat}")

    # 计算相对误差
    relative_errors = np.abs((x0 - x0_hat) / x0) * 100
    print(f"相对误差(%): {relative_errors}")
    print(f"平均相对误差: {relative_errors.mean():.2f}%")

    # ====== 第六步：预测未来值 ======
    predict_values = []
    for k in range(n, n + predict_steps):
        pred = predict_x1(k) - predict_x1(k - 1)
        predict_values.append(pred)

    print(f"\n未来 {predict_steps} 步预测值: {predict_values}")

    return predict_values, a, b, relative_errors


# ====== 示例：预测某城市未来3年的GDP ======
# 已知过去6年的GDP数据（亿元）
gdp_data = [120, 135, 150, 170, 195, 220]

predictions, a, b, errors = gm11(gdp_data, predict_steps=3)

# 打印预测结果
print("\n=== GDP预测结果 ===")
years_known = list(range(2018, 2024))
years_predict = list(range(2024, 2027))

print("年份    实际值    拟合值    误差(%)")
for i, year in enumerate(years_known):
    fitted = predictions[0] if i == 0 else None  # placeholder
    print(f"{year}  {gdp_data[i]:>8.1f}")

# 重新计算拟合值用于展示
x1_pred = np.cumsum(gdp_data)
fitted = [gdp_data[0]]
for k in range(1, len(gdp_data)):
    val = (gdp_data[0] - b/a) * np.exp(-a*k) + b/a - ((gdp_data[0] - b/a) * np.exp(-a*(k-1)) + b/a)
    fitted.append(val)

print("\n年份    实际值    拟合值    误差(%)")
for i, year in enumerate(years_known):
    err = abs((gdp_data[i] - fitted[i]) / gdp_data[i]) * 100
    print(f"{year}  {gdp_data[i]:>8.1f}  {fitted[i]:>8.1f}  {err:>6.2f}")

print("\n预测年份  预测值")
for year, pred in zip(years_predict, predictions):
    print(f"{year}    {pred:>8.1f}")
```

> **小贴士**：灰色预测适用于数据量少（4~10个点）、呈近似指数增长的趋势。如果数据波动很大或呈下降趋势，GM(1,1) 效果不好，可以考虑用 GM(2,1) 或其他模型。

```python
import numpy as np
import matplotlib.pyplot as plt

# ========================================
# 时间序列预测（简单移动平均 + 指数平滑）
# ========================================

print("=== 时间序列预测 ===")

# 生成模拟数据：某商品月销量（含趋势和季节性）
np.random.seed(42)
n_months = 36  # 3年的月度数据

# 趋势：每月增长2
trend = np.arange(n_months) * 2

# 季节性：12个月周期
seasonal = 10 * np.sin(2 * np.pi * np.arange(n_months) / 12)

# 随机噪声
noise = np.random.normal(0, 3, n_months)

# 最终数据
sales = 50 + trend + seasonal + noise

for i in range(0, n_months, 6):
    print(f"  第{i+1:2d}月: {sales[i]:.1f}")

# ====== 方法一：简单移动平均（SMA） ======
def simple_moving_average(data, window=3):
    """简单移动平均"""
    sma = np.full(len(data), np.nan)
    for i in range(window - 1, len(data)):
        sma[i] = data[i - window + 1: i + 1].mean()
    return sma

sma_3 = simple_moving_average(sales, window=3)
sma_6 = simple_moving_average(sales, window=6)

# ====== 方法二：指数平滑（EMA） ======
def exponential_smoothing(data, alpha=0.3):
    """
    一次指数平滑
    alpha: 平滑系数（0~1），越大越关注近期数据
    """
    smoothed = np.zeros(len(data))
    smoothed[0] = data[0]
    for i in range(1, len(data)):
        smoothed[i] = alpha * data[i] + (1 - alpha) * smoothed[i - 1]
    return smoothed

ema_03 = exponential_smoothing(sales, alpha=0.3)
ema_07 = exponential_smoothing(sales, alpha=0.7)

# ====== 方法三：二次指数平滑（含趋势） ======
def double_exponential_smoothing(data, alpha=0.3, beta=0.1):
    """
    二次指数平滑（Holt法）
    适用于含趋势的时间序列
    """
    n = len(data)
    level = np.zeros(n)    # 水平分量
    trend = np.zeros(n)    # 趋势分量

    level[0] = data[0]
    trend[0] = data[1] - data[0]

    for i in range(1, n):
        level[i] = alpha * data[i] + (1 - alpha) * (level[i - 1] + trend[i - 1])
        trend[i] = beta * (level[i] - level[i - 1]) + (1 - beta) * trend[i - 1]

    return level, trend

level, trend = double_exponential_smoothing(sales, alpha=0.3, beta=0.1)

# 预测未来6个月

forecast = np.zeros(forecast_steps)
for h in range(forecast_steps):
    forecast[h] = level[-1] + (h + 1) * trend[-1]

print(f"\n二次指数平滑预测未来{forecast_steps}个月:")

print(f"  各簇样本数: {np.bincount(labels)}")
print(f"  簇中心:\n{kmeans.cluster_centers_}")
print(f"  轮廓系数: {silhouette_score(X, labels):.4f}")

# ====== 3. 可视化聚类结果 ======
plt.figure(figsize=(8, 6))
colors = ['red', 'blue', 'green', 'purple', 'orange']

for i in range(best_k):
    cluster_points = X[labels == i]
    plt.scatter(cluster_points[:, 0], cluster_points[:, 1],
                c=colors[i], label=f'Cluster {i}', alpha=0.6, edgecolors='k')

# 画出簇中心
plt.scatter(kmeans.cluster_centers_[:, 0], kmeans.cluster_centers_[:, 1],
            c='yellow', marker='*', s=300, edgecolors='black', linewidth=2,
            label='Centroids', zorder=5)

plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title(f'KMeans Clustering (K={best_k}, Silhouette={silhouette_score(X, labels):.3f})')
plt.legend()
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig('kmeans_result.png', dpi=150)
plt.show()
print("聚类结果图已保存为 kmeans_result.png")
```

> **小贴士**：确定最佳聚类数K的两种方法：
> 1. **肘部法则**：看SSE随K变化的曲线，在"肘部"处的K值最佳
> 2. **轮廓系数**：取值范围[-1, 1]，越接近1说明聚类效果越好

### 4.2 微分方程求解——SIR传染病模型

```python
import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import odeint

# ========================================
# SIR 传染病模型（微分方程求解）
# ========================================

# SIR模型将人群分为三类：
# S (Susceptible)：易感者——可能被感染的人
# I (Infected)：感染者——已经患病的人
# R (Recovered)：康复者——已康复（或死亡）的人

def sir_model(y, t, beta, gamma):
    """
    SIR模型的微分方程组

    参数:
        y: [S, I, R] 当前各群体人数
        t: 时间（odeint会自动传入）
        beta: 感染率
        gamma: 康复率

    返回:
        [dS/dt, dI/dt, dR/dt]
    """
    S, I, R = y
    N = S + I + R  # 总人口

    # dS/dt = -beta * S * I / N
    dSdt = -beta * S * I / N

    # dI/dt = beta * S * I / N - gamma * I
    dIdt = beta * S * I / N - gamma * I

    # dR/dt = gamma * I
    dRdt = gamma * I

    return [dSdt, dIdt, dRdt]


print("=== SIR 传染病模型 ===")

# 初始条件
N = 10000      # 总人口
I0 = 10        # 初始感染者
R0 = 0         # 初始康复者
S0 = N - I0 - R0  10  # 初始易感者

# 参数
beta = 0.5     # 感染率（每个感染者每天接触的人数 × 感染概率）
gamma = 0.1    # 康复率（每天康复的比例）
R0_basic = beta / gamma  # 基本再生数

print(f"总人口: {N}")
print(f"初始感染者: {I0}")
print(f"感染率 β = {beta}")
print(f"康复率 γ = {gamma}")
print(f"基本再生数 R0 = β/γ = {R0_basic:.2f}")
if R0_basic > 1:
    print("→ R0 > 1，传染病会扩散！")
else:
    print("→ R0 ≤ 1，传染病会自行消亡。")

# 时间点（模拟180天）
t = np.linspace(0, 180, 1000)

# 初始状态
y0 = [S0, I0, R0]

# 求解微分方程
solution = odeint(sir_model, y0, t, args=(beta, gamma))
S = solution[:, 0]
I = solution[:, 1]
R = solution[:, 2]

# 找到感染高峰
peak_idx = np.argmax(I)
print(f"\n感染高峰:")
print(f"  第 {t[peak_idx]:.0f} 天达到峰值")
print(f"  峰值感染人数: {I[peak_idx]:.0f}")
print(f"  最终累计感染: {R[-1]:.0f} ({R[-1]/N*100:.1f}%)")

# ====== 可视化 ======
plt.figure(figsize=(12, 6))

plt.fill_between(t, 0, S, alpha=0.3, color='blue', label='Susceptible (S)')
plt.fill_between(t, S, S + I, alpha=0.5, color='red', label='Infected (I)')
plt.fill_between(t, S + I, S + I + R, alpha=0.3, color='green', label='Recovered (R)')

plt.plot(t, S, 'b-', linewidth=2)
plt.plot(t, I, 'r-', linewidth=2)
plt.plot(t, R, 'g-', linewidth=2)

plt.axvline(x=t[peak_idx], color='gray', linestyle='--', alpha=0.5)
plt.annotate(f'Peak: Day {t[peak_idx]:.0f}\nI={I[peak_idx]:.0f}',
             xy=(t[peak_idx], I[peak_idx]),
             xytext=(t[peak_idx] + 20, I[peak_idx] + 200),
             arrowprops=dict(arrowstyle='->', color='black'),
             fontsize=10)

plt.xlabel('Days')
plt.ylabel('Number of People')
plt.title(f'SIR Model (β={beta}, γ={gamma}, R0={R0_basic:.1f})')
plt.legend(loc='upper right')
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig('sir_model.png', dpi=150)
plt.show()
print("SIR模型图已保存为 sir_model.png")

# ====== 对比不同参数下的传播情况 ======
print("\n=== 不同参数对比 ===")

plt.figure(figsize=(12, 5))

# 对比不同beta值
plt.subplot(1, 2, 1)
for beta_val in [0.2, 0.5, 0.8, 1.2]:
    sol = odeint(sir_model, y0, t, args=(beta_val, gamma))
    plt.plot(t, sol[:, 1], label=f'β={beta_val} (R0={beta_val/gamma:.1f})')

plt.xlabel('Days')
plt.ylabel('Infected')
plt.title('Effect of β (Infection Rate)')
plt.legend()
plt.grid(True, alpha=0.3)

# 对比不同gamma值
plt.subplot(1, 2, 2)
for gamma_val in [0.05, 0.1, 0.2, 0.3]:
    sol = odeint(sir_model, y0, t, args=(beta, gamma_val))
    plt.plot(t, sol[:, 1], label=f'γ={gamma_val} (R0={beta/gamma_val:.1f})')

plt.xlabel('Days')
plt.ylabel('Infected')
plt.title('Effect of γ (Recovery Rate)')
plt.legend()
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('sir_parameter_comparison.png', dpi=150)
plt.show()
print("参数对比图已保存为 sir_parameter_comparison.png")
```

> **小贴士**：`scipy.integrate.odeint` 是求解常微分方程组的标准工具。使用步骤：
> 1. 定义微分方程函数 `f(y, t, params)`
> 2. 设置初始条件 `y0`
> 3. 设置时间点 `t`
> 4. 调用 `odeint(f, y0, t, args=params)` 求解

## 附录：算法选择速查表

| 问题类型 | 推荐算法 | 适用场景 |
|----------|----------|----------|
| 确定指标权重 | AHP / 熵权法 | 主观赋权用AHP，客观赋权用熵权法 |
| 方案排序 | TOPSIS / 模糊综合评价 | 定量数据用TOPSIS，定性数据用模糊评价 |
| 资源分配 | 线性规划 | 目标和约束都是线性的 |
| 选址/选方案 | 整数规划 | 变量必须为整数 |
| 路径优化 | 遗传算法 / TSP | 组合优化问题 |
| 少数据预测 | 灰色预测 GM(1,1) | 数据量少于10个 |
| 因果预测 | 线性/多元回归 | 有多个自变量 |
| 时间序列 | 指数平滑 / ARIMA | 有历史时间序列数据 |
| 客户分群 | KMeans聚类 | 无标签数据分组 |
| 动态系统 | 微分方程 | 系统状态随时间连续变化 |

> **避坑指南**：没有"万能模型"。选择模型时，先看数据特征（数据量多少、是否有标签、是否有时序性），再看问题类型（评价/优化/预测/分类），最后选择合适的算法。
