## 一、matplotlib 基础

matplotlib 是 Python 最基础的绘图库。掌握它，你就能画出绝大多数需要的图表。

### 1.1 基础设置与中文支持

```python
import matplotlib.pyplot as plt
import numpy as np

# ====== 解决中文显示问题（必加！） ======
plt.rcParams['font.sans-serif'] = ['SimHei']  # Windows 用黑体
# plt.rcParams['font.sans-serif'] = ['Arial Unicode MS']  # macOS 用这个
plt.rcParams['axes.unicode_minus'] = False  # 解决负号显示为方块的问题

# ====== 通用设置 ======
plt.rcParams['figure.figsize'] = (8, 5)    # 默认图片大小
plt.rcParams['figure.dpi'] = 100            # 默认分辨率
plt.rcParams['font.size'] = 12              # 默认字体大小
```

> **避坑指南**：中文显示乱码是 matplotlib 最常见的问题。上面的代码是 Windows 下的解决方案。如果还是乱码，试试：
> ```python
> import matplotlib
> matplotlib.font_manager.fontManager.addfont('C:/Windows/Fonts/simhei.ttf')
> plt.rcParams['font.sans-serif'] = ['SimHei']
> ```

### 1.2 折线图

```python
import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例：某城市月平均气温变化 ======
months = ['1月', '2月', '3月', '4月', '5月', '6月',
          '7月', '8月', '9月', '10月', '11月', '12月']
temp_2023 = [2, 5, 12, 18, 23, 28, 31, 30, 25, 19, 11, 4]
temp_2022 = [3, 6, 11, 17, 22, 27, 30, 29, 24, 18, 10, 3]

fig, ax = plt.subplots(figsize=(10, 5))

# 画两条折线
ax.plot(months, temp_2023, 'b-o', linewidth=2, markersize=6,
        label='2023年', markerfacecolor='white', markeredgewidth=2)
ax.plot(months, temp_2022, 'r--s', linewidth=2, markersize=6,
        label='2022年', markerfacecolor='white', markeredgewidth=2)

# 设置标题和标签
ax.set_title('某城市月平均气温变化', fontsize=16, fontweight='bold', pad=15)
ax.set_xlabel('月份', fontsize=13)
ax.set_ylabel('气温 (°C)', fontsize=13)

# 设置图例
ax.legend(loc='upper right', fontsize=11, framealpha=0.9)

# 添加网格
ax.grid(True, alpha=0.3, linestyle='--')

# 设置Y轴范围
ax.set_ylim(0, 35)

# 在每个点上方标注数值
for i, (t23, t22) in enumerate(zip(temp_2023, temp_2022)):
    ax.annotate(f'{t23}', (i, t23), textcoords="offset points",
                xytext=(0, 10), ha='center', fontsize=8, color='blue')

plt.tight_layout()
plt.savefig('line_chart.png', dpi=300)
plt.show()
print("折线图已保存为 line_chart.png")
```

> **小贴士**：`plot` 函数的格式字符串：第一个字母是颜色（b=蓝, r=红, g=绿, k=黑），第二个是线型（-=实线, --=虚线, :=点线），第三个是标记（o=圆点, s=方块, ^=三角）。

### 1.3 柱状图

```python
import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例1：分组柱状图 ======
# 各班级各科平均分对比
classes = ['A班', 'B班', 'C班']
subjects = ['数学', '英语', '编程']
scores = {
    '数学': [85, 78, 92],
    '英语': [82, 88, 85],
    '编程': [90, 76, 88]
}

x = np.arange(len(classes))  # 班级位置
width = 0.25  # 柱子宽度

fig, ax = plt.subplots(figsize=(9, 5))

# 为每个科目画一组柱子
colors = ['#2196F3', '#FF9800', '#4CAF50']
for i, (subject, values) in enumerate(scores.items()):
    bars = ax.bar(x + i * width, values, width, label=subject, color=colors[i])
    # 在柱子上方标注数值
    for bar in bars:
        height = bar.get_height()
        ax.annotate(f'{height}',
                    xy=(bar.get_x() + bar.get_width() / 2, height),
                    xytext=(0, 3), textcoords="offset points",
                    ha='center', va='bottom', fontsize=10)

ax.set_title('各班级各科平均分对比', fontsize=15, fontweight='bold')
ax.set_xlabel('班级', fontsize=13)
ax.set_ylabel('分数', fontsize=13)
ax.set_xticks(x + width)
ax.set_xticklabels(classes)
ax.legend(loc='upper right')
ax.set_ylim(0, 105)
ax.grid(True, axis='y', alpha=0.3, linestyle='--')

plt.tight_layout()
plt.savefig('bar_chart.png', dpi=300)
plt.show()

# ====== 示例2：堆叠柱状图 ======
fig, ax = plt.subplots(figsize=(8, 5))

# 各季度收入构成
quarters = ['Q1', 'Q2', 'Q3', 'Q4']
product_a = [25, 30, 35, 40]
product_b = [15, 20, 25, 30]
product_c = [10, 15, 20, 25]

ax.bar(quarters, product_a, label='产品A', color='#2196F3')
ax.bar(quarters, product_b, bottom=product_a, label='产品B', color='#FF9800')
ax.bar(quarters, product_c,
       bottom=np.array(product_a) + np.array(product_b),
       label='产品C', color='#4CAF50')

# 标注总数
totals = np.array(product_a) + np.array(product_b) + np.array(product_c)
for i, total in enumerate(totals):
    ax.annotate(f'{total}', (i, total), textcoords="offset points",
                xytext=(0, 5), ha='center', fontsize=11, fontweight='bold')

ax.set_title('各季度收入构成（堆叠柱状图）', fontsize=15, fontweight='bold')
ax.set_xlabel('季度', fontsize=13)
ax.set_ylabel('收入（万元）', fontsize=13)
ax.legend(loc='upper left')
ax.grid(True, axis='y', alpha=0.3, linestyle='--')

plt.tight_layout()
plt.savefig('stacked_bar.png', dpi=300)
plt.show()
print("柱状图已保存")
```

## 二、seaborn 进阶

seaborn 基于 matplotlib，能画出更美观的统计图表，且代码更简洁。

### 2.1 热力图

热力图在数学建模中非常常用，特别适合展示相关系数矩阵。

```python
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例：各指标相关性热力图 ======
np.random.seed(42)

# 生成模拟数据
data = pd.DataFrame({
    'GDP': np.random.uniform(1000, 10000, 50),
    '人口': np.random.uniform(100, 1000, 50),
    '面积': np.random.uniform(500, 5000, 50),
    '教育投入': np.random.uniform(50, 500, 50),
    '医疗投入': np.random.uniform(30, 300, 50)
})

# 让一些变量相关
data['教育投入'] = data['GDP'] * 0.05 + np.random.normal(0, 20, 50)
data['医疗投入'] = data['GDP'] * 0.03 + data['人口'] * 0.1 + np.random.normal(0, 15, 50)

# 计算相关系数矩阵
corr_matrix = data.corr()

print("相关系数矩阵:")
print(corr_matrix.round(3))

# 画热力图
fig, ax = plt.subplots(figsize=(8, 6))

sns.heatmap(
    corr_matrix,
    annot=True,            # 在格子中显示数值
    fmt='.2f',             # 数值格式
    cmap='RdBu_r',         # 颜色方案（红蓝反转）
    center=0,              # 颜色中心设为0
    square=True,           # 正方形格子
    linewidths=0.5,        # 格子间线宽
    cbar_kws={'shrink': 0.8, 'label': '相关系数'},
    vmin=-1, vmax=1,       # 颜色范围
    ax=ax
)

ax.set_title('各指标相关性热力图', fontsize=15, fontweight='bold', pad=15)
plt.tight_layout()
plt.savefig('heatmap.png', dpi=300)
plt.show()
print("热力图已保存为 heatmap.png")
```

> **小贴士**：热力图常用的颜色方案：
> - `RdBu_r`：红蓝反转（相关性最常用，正相关红色，负相关蓝色）
> - `YlOrRd`：黄橙红（适合全是正值的数据）
> - `coolwarm`：冷暖色

```python
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

plt.rcParams['font.sans-serif'] = ['SimHei']

axes[0].set_ylabel('成绩', fontsize=12)
axes[0].grid(True, axis='y', alpha=0.3)

# 子图2：分组箱线图（班级 × 科目）
sns.boxplot(data=data, x='班级', y='成绩', hue='科目', ax=axes[1],
            palette='Set3')
axes[1].set_title('各班各科成绩分布', fontsize=14, fontweight='bold')
axes[1].set_ylabel('成绩', fontsize=12)
axes[1].legend(title='科目')
axes[1].grid(True, axis='y', alpha=0.3)

plt.tight_layout()
plt.savefig('boxplot.png', dpi=300)
plt.show()
print("箱线图已保存为 boxplot.png")
```

> **小贴士**：箱线图能一眼看出数据的分布特征：中位数（箱体中间的线）、四分位数（箱体上下边）、异常值（圆点）。非常适合比较不同组的数据分布。

### 2.3 pairplot（散点图矩阵）

```python
import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例：多变量关系散点图矩阵 ======
np.random.seed(42)

# 生成鸢尾花式数据
n = 100
data = pd.DataFrame({
    '花萼长度': np.concatenate([np.random.normal(5, 0.5, n//3),
                                np.random.normal(6, 0.6, n//3),
                                np.random.normal(6.5, 0.7, n//3)]),
    '花萼宽度': np.concatenate([np.random.normal(3.5, 0.3, n//3),
                                np.random.normal(2.8, 0.4, n//3),
                                np.random.normal(3.0, 0.5, n//3)]),
    '花瓣长度': np.concatenate([np.random.normal(1.5, 0.2, n//3),
                                np.random.normal(4.5, 0.5, n//3),
                                np.random.normal(5.5, 0.6, n//3)]),
    '花瓣宽度': np.concatenate([np.random.normal(0.2, 0.1, n//3),
                                np.random.normal(1.5, 0.2, n//3),
                                np.random.normal(2.0, 0.3, n//3)]),
    '品种': ['Setosa']*33 + ['Versicolor']*33 + ['Virginica']*34
})

# 画散点图矩阵
g = sns.pairplot(
    data,
    hue='品种',              # 按品种着色
    diag_kind='kde',         # 对角线画核密度图
    palette='Set1',
    markers=['o', 's', 'D'], # 不同品种用不同标记
    height=2.5,
    plot_kws={'alpha': 0.6, 's': 30, 'edgecolor': 'k'}
)

g.fig.suptitle('鸢尾花多变量关系散点图矩阵', fontsize=16, fontweight='bold', y=1.02)
plt.tight_layout()
plt.savefig('pairplot.png', dpi=300, bbox_inches='tight')
plt.show()
print("散点图矩阵已保存为 pairplot.png")
```

> **小贴士**：`pairplot` 一行代码就能画出所有变量两两之间的散点图，是探索性数据分析（EDA）的利器。对角线可以显示各变量的分布（直方图或核密度图）。

### 3.1 拟合曲线图

```python
import matplotlib.pyplot as plt
import numpy as np
from scipy.optimize import curve_fit

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例：非线性拟合 ======
np.random.seed(42)

# 生成数据：y = a * exp(b*x) + noise
x_data = np.linspace(0, 5, 30)
y_true = 2 * np.exp(0.5 * x_data)
y_data = y_true + np.random.normal(0, 2, 30)

# 定义拟合函数
def exp_func(x, a, b):
    return a * np.exp(b * x)

# 拟合
popt, pcov = curve_fit(exp_func, x_data, y_data, p0=[2, 0.5])
a_fit, b_fit = popt

# 计算R²
y_fit = exp_func(x_data, *popt)
ss_res = np.sum((y_data - y_fit) ** 2)
ss_tot = np.sum((y_data - y_data.mean()) ** 2)
r2 = 1 - ss_res / ss_tot

print(f"拟合结果: a={a_fit:.4f}, b={b_fit:.4f}")
print(f"R² = {r2:.4f}")

# 画图
fig, ax = plt.subplots(figsize=(9, 6))

# 原始数据散点
ax.scatter(x_data, y_data, c='blue', s=60, alpha=0.6,
           edgecolors='navy', label='原始数据', zorder=3)

# 拟合曲线
x_smooth = np.linspace(0, 5, 200)
y_smooth = exp_func(x_smooth, *popt)
ax.plot(x_smooth, y_smooth, 'r-', linewidth=2.5,
        label=f'拟合曲线: y={a_fit:.2f}·e^({b_fit:.2f}x)\nR²={r2:.4f}', zorder=2)

# 置信区间（简化版：用残差标准差）
residuals = y_data - y_fit
std_err = np.std(residuals)
ax.fill_between(x_smooth, y_smooth - 2*std_err, y_smooth + 2*std_err,
                alpha=0.15, color='red', label='95%置信区间')

ax.set_title('指数拟合曲线', fontsize=15, fontweight='bold')
ax.set_xlabel('x', fontsize=13)
ax.set_ylabel('y', fontsize=13)
ax.legend(fontsize=11, loc='upper left')
ax.grid(True, alpha=0.3, linestyle='--')
ax.set_xlim(-0.2, 5.2)

plt.tight_layout()
plt.savefig('fit_curve.png', dpi=300)
plt.show()
print("拟合曲线图已保存为 fit_curve.png")
```

### 3.2 雷达图（多指标评价）

```python
import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例：多城市综合评价雷达图 ======
categories = ['经济', '教育', '医疗', '环境', '交通', '文化']
N = len(categories)

# 三个城市各维度得分（0-100）
cities = {
    '城市A': [85, 78, 82, 70, 88, 75],
    '城市B': [75, 90, 85, 80, 72, 88],
    '城市C': [90, 85, 75, 92, 80, 82]
}

# 计算角度
angles = np.linspace(0, 2 * np.pi, N, endpoint=False).tolist()
angles += angles[:1]  # 闭合

fig, ax = plt.subplots(figsize=(8, 8), subplot_kw=dict(polar=True))

colors = ['blue', 'red', 'green']
for idx, (city, values) in enumerate(cities.items()):
    values_closed = values + values[:1]  # 闭合
    ax.plot(angles, values_closed, 'o-', linewidth=2,
            label=city, color=colors[idx], markersize=6)
    ax.fill(angles, values_closed, alpha=0.15, color=colors[idx])

# 设置标签
ax.set_xticks(angles[:-1])
ax.set_xticklabels(categories, fontsize=13)

# 设置径向轴
ax.set_ylim(0, 100)
ax.set_yticks([20, 40, 60, 80, 100])
ax.set_yticklabels(['20', '40', '60', '80', '100'], fontsize=10)

ax.set_title('城市综合发展水平雷达图', fontsize=16, fontweight='bold', pad=20)
ax.legend(loc='upper right', bbox_to_anchor=(1.3, 1.1), fontsize=12)
ax.grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('radar_chart.png', dpi=300, bbox_inches='tight')
plt.show()
print("雷达图已保存为 radar_chart.png")
```

> **小贴士**：雷达图是多指标评价的标配图表。画图关键是把数据"闭合"（首尾相接），然后用 `polar=True` 创建极坐标轴。

### 3.3 甘特图

```python
import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 示例：数学建模比赛进度甘特图 ======
tasks = [
    '读题与讨论',
    '查阅文献',
    '数据收集与预处理',
    '模型建立',
    '模型编程实现',
    '结果分析与验证',
    '论文撰写',
    '论文修改与排版',
    '最终检查'
]
```

### 4.1 统一美化模板

```python
import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

def setup_axes(ax, title, xlabel, ylabel):
    """统一的坐标轴美化函数"""
    # 标题
    ax.set_title(title, fontsize=15, fontweight='bold', pad=15)
    # 轴标签
    ax.set_xlabel(xlabel, fontsize=13)
    ax.set_ylabel(ylabel, fontsize=13)
    # 去掉上边和右边的边框
    ax.spines['top'].set_visible(False)
    ax.spines['right'].set_visible(False)
    # 网格
    ax.grid(True, alpha=0.3, linestyle='--')
    # 刻度标签大小
    ax.tick_params(axis='both', labelsize=11)


# ====== 美化前后对比 ======
x = np.linspace(0, 10, 50)
y1 = np.sin(x)
y2 = np.cos(x)

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# 左图：未美化
axes[0].plot(x, y1, x, y2)
axes[0].set_title('sin & cos')
axes[0].set_xlabel('x')
axes[0].set_ylabel('y')

# 右图：美化后
axes[1].plot(x, y1, 'b-o', linewidth=2, markersize=4, label='sin(x)',
             markerfacecolor='white', markeredgewidth=1.5)
axes[1].plot(x, y2, 'r-s', linewidth=2, markersize=4, label='cos(x)',
             markerfacecolor='white', markeredgewidth=1.5)
setup_axes(axes[1], '正弦与余弦函数', 'x', 'y')
axes[1].legend(fontsize=12, framealpha=0.9, loc='upper right')

plt.tight_layout()
plt.savefig('beautify_comparison.png', dpi=300)
plt.show()
```

## 五、导出高清图片

```python
import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False

# ====== 画一个示例图 ======
fig, ax = plt.subplots(figsize=(8, 5))
x = np.linspace(0, 10, 100)
ax.plot(x, np.sin(x), 'b-', linewidth=2, label='sin(x)')
ax.plot(x, np.cos(x), 'r--', linewidth=2, label='cos(x)')
ax.set_title('高清导出示例', fontsize=15, fontweight='bold')
ax.legend(fontsize=12)
ax.grid(True, alpha=0.3)

# ====== 导出方法 ======

# 方法1：导出为 PNG（最常用，dpi=300 足够清晰）
plt.savefig('output_png.png', dpi=300, bbox_inches='tight')
# bbox_inches='tight' 会自动裁剪多余白边

# 方法2：导出为 PDF（矢量图，无限缩放不失真，论文推荐）
plt.savefig('output_pdf.pdf', bbox_inches='tight')

# 方法3：导出为 SVG（矢量图，可编辑）
plt.savefig('output_svg.svg', bbox_inches='tight')

# 方法4：导出为高分辨率PNG（dpi=600，打印级）
plt.savefig('output_hd.png', dpi=600, bbox_inches='tight')

plt.show()

print("=== 导出说明 ===")
print("PNG  (dpi=300): 适合论文插图，文件适中")
print("PNG  (dpi=600): 适合打印，文件较大")
print("PDF  (矢量):    最佳选择，缩放不失真")
print("SVG  (矢量):    可用Illustrator编辑")
print()
print("关键参数说明：")
print("  dpi: 分辨率（每英寸点数），300以上即可")
print("  bbox_inches='tight': 自动裁剪白边")
print("  facecolor='white': 确保背景为白色")
```

> **避坑指南**：论文中图表一定要导出为高分辨率（至少 dpi=300）或矢量格式（PDF/SVG）。低分辨率的位图放大后会模糊，给评委留下不专业的印象。

## 六、竞赛论文配图检查清单

在提交论文前，请逐项检查每一张图表：

### 基本要求
- [ ] **标题**：每张图都有简洁明了的标题
- [ ] **坐标轴标签**：横轴和纵轴都有标签和单位
- [ ] **图例**：有多条线/多个数据系列时有图例
- [ ] **字体大小**：标题、标签、图例的字体大小适中（不小不挤）
- [ ] **分辨率**：至少 dpi=300，或导出为矢量图

### 数据准确性
- [ ] **数据正确**：图表中的数据与正文描述一致
- [ ] **无遗漏**：关键结果都有图表展示
- [ ] **比例合理**：坐标轴范围合适，不会让数据"挤在一起"或"太分散"

### 美观性
- [ ] **中文正常**：所有中文正常显示，无方块乱码
- [ ] **颜色协调**：颜色搭配美观，不过于花哨
- [ ] **网格线**：有淡淡的网格线辅助阅读
- [ ] **无多余白边**：图片紧凑，无大面积空白
- [ ] **标注清晰**：关键数据点有数值标注

### 论文规范
- [ ] **图编号**：每张图有编号（图1、图2...）
- [ ] **图题位置**：图题在图下方
- [ ] **正文中引用**：正文中提到"如图X所示"
- [ ] **格式统一**：所有图表风格统一（字体、颜色、大小）
- [ ] **文件格式**：使用 PNG(dpi=300) 或 PDF 格式

### 常见图表类型检查
| 图表类型 | 检查要点 |
|----------|----------|
| 折线图 | 线条清晰，标记点不重叠 |
| 柱状图 | 柱子间距合理，数值标注可读 |
| 散点图 | 点不过于密集，有透明度 |
| 饼图 | 扇区比例正确，标签不重叠 |
| 热力图 | 颜色方案合理，有数值标注 |
| 雷达图 | 各轴比例一致，数据闭合 |
