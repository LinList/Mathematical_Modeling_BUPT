**案例背景**：某城市交通管理部门希望优化出租车调度策略，减少乘客等待时间，同时提高出租车利用率。我们需要分析现有数据，建立评价模型和优化模型，给出调度建议。

## 第一步：数据生成与预处理

在真实比赛中，数据通常由赛方提供。这里我们生成模拟数据来演示流程。

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.optimize import linprog
from sklearn.preprocessing import MinMaxScaler
from sklearn.cluster import KMeans

plt.rcParams['axes.unicode_minus'] = False
plt.rcParams['figure.dpi'] = 100
np.random.seed(42)

# ========================================
# 第一步：生成模拟数据
# ========================================
print("=" * 60)
print("第一步：数据生成与预处理")
print("=" * 60)

# 模拟某城市6个区域的出租车运营数据
# 区域：商业区、住宅区A、住宅区B、工业区、火车站、机场
zones = ['商业区', '住宅区A', '住宅区B', '工业区', '火车站', '机场']
n_days = 30  # 30天数据

# 生成每日数据
data_list = []
for day in range(1, n_days + 1):
    for zone in zones:
        # 生成各区域每日指标
        base_demand = {
            '商业区': 500, '住宅区A': 200, '住宅区B': 180,
            '工业区': 150, '火车站': 400, '机场': 350
        }[zone]

        # 需求量（加随机波动）
        demand = base_demand + np.random.normal(0, base_demand * 0.15)

        # 可用出租车数量
        available_taxis = int(demand * np.random.uniform(0.6, 1.2))

        # 平均等待时间（分钟）
        if available_taxis > 0:
            wait_time = max(2, (demand / available_taxis) * 5 + np.random.normal(0, 3))
        else:
            wait_time = 30

        # 平均行程距离（公里）
        avg_distance = {
            '商业区': 5.2, '住宅区A': 8.5, '住宅区B': 9.0,
            '工业区': 12.3, '火车站': 6.8, '机场': 25.5
        }[zone] + np.random.normal(0, 1)

        # 收入（元）
        revenue = avg_distance * 2.3 * available_taxis * np.random.uniform(0.7, 1.0)

        # 乘客满意度（1-10分），与等待时间负相关
        satisfaction = max(1, min(10, 10 - wait_time / 4 + np.random.normal(0, 0.5)))

        data_list.append({
            '日期': f'2024-06-{day:02d}',
            '区域': zone,
            '乘客需求量': int(demand),
            '可用出租车数': available_taxis,
            '平均等待时间': round(wait_time, 1),
            '平均行程距离': round(avg_distance, 1),
            '日收入': round(revenue, 0),
            '满意度评分': round(satisfaction, 1)
        })

# 创建DataFrame
df = pd.DataFrame(data_list)

print("\n--- 数据概览 ---")
print(f"数据形状: {df.shape}（{df.shape[0]}行 × {df.shape[1]}列）")
print(f"日期范围: {df['日期'].min()} ~ {df['日期'].max()}")
print(f"区域数量: {df['区域'].nunique()}")
print(f"\n前10行数据:")
print(df.head(10))
print(f"\n数据信息:")
print(df.info())
print(f"\n统计描述:")
print(df.describe().round(2))

# ====== 数据预处理 ======
print("\n--- 数据预处理 ---")

# 检查缺失值
print(f"缺失值统计:\n{df.isnull().sum()}")

# 检查重复行
print(f"\n重复行数: {df.duplicated().sum()}")

# 添加派生特征
df['供需比'] = df['可用出租车数'] / df['乘客需求量']
df['单车收入'] = df['日收入'] / df['可用出租车数']
df['利用率'] = (df['乘客需求量'] / df['可用出租车数']).clip(upper=2.0)  # 超过1表示供不应求

print(f"\n添加派生特征后:")
print(df[['区域', '乘客需求量', '可用出租车数', '供需比', '利用率', '单车收入']].head(10))

# 保存预处理后的数据
df.to_csv('taxi_data_cleaned.csv', index=False, encoding='utf-8-sig')
print("\n预处理数据已保存为 taxi_data_cleaned.csv")
```

## 第三步：模型实现

### 模型一：基于TOPSIS的区域运营评价

```python
# ========================================
# 第三步：模型实现
# ========================================
print("\n" + "=" * 60)
print("第三步：模型实现")
print("=" * 60)

# ====== 模型一：TOPSIS评价各区域运营效率 ======
print("\n--- 模型一：TOPSIS区域运营评价 ---")

# 汇总各区域30天的平均数据
zone_data = df.groupby('区域').agg({
    '乘客需求量': 'mean',
    '可用出租车数': 'mean',
    '平均等待时间': 'mean',
    '平均行程距离': 'mean',
    '日收入': 'mean',
    '满意度评分': 'mean'
}).round(2)

print("\n各区域汇总数据:")
print(zone_data)

# 选择评价指标
# 效益型（越大越好）：需求量、日收入、满意度
# 成本型（越小越好）：等待时间
criteria_names = ['需求量', '可用车辆', '等待时间', '行程距离', '日收入', '满意度']
criteria_types = ['benefit', 'benefit', 'cost', 'benefit', 'benefit', 'benefit']

decision_matrix = zone_data.values

# 熵权法确定权重
def entropy_weight(matrix, types):
    """熵权法"""
    m = matrix.copy().astype(float)
    n_schemes, n_criteria = m.shape

    # 正向化
    for j in range(n_criteria):
        if types[j] == 'cost':
            m[:, j] = m[:, j].max() - m[:, j]

    # 标准化
    col_sums = m.sum(axis=0)
    normalized = m / col_sums

    # 信息熵
    k = 1.0 / np.log(n_schemes)
    entropy = np.zeros(n_criteria)
    for j in range(n_criteria):
        p = normalized[:, j]
        p_nonzero = p[p > 0]
        entropy[j] = -k * np.sum(p_nonzero * np.log(p_nonzero))

    # 权重
    d = 1 - entropy
    weights = d / d.sum()
    return weights

weights = entropy_weight(decision_matrix, criteria_types)
print(f"\n熵权法权重:")
for name, w in zip(criteria_names, weights):
    print(f"  {name}: {w:.4f}")

# TOPSIS评价
def topsis_evaluate(matrix, weights, types):
    """TOPSIS法"""
    m = matrix.copy().astype(float)
    n_criteria = m.shape[1]

    # 正向化
    for j in range(n_criteria):
        if types[j] == 'cost':
            m[:, j] = m[:, j].max() - m[:, j]

    # 标准化
    col_norms = np.sqrt((m ** 2).sum(axis=0))
    standard = m / col_norms

    # 加权
    weighted = standard * weights

    # 正负理想解
    pos_ideal = weighted.max(axis=0)
    neg_ideal = weighted.min(axis=0)

    # 距离
    d_pos = np.sqrt(((weighted - pos_ideal) ** 2).sum(axis=1))
    d_neg = np.sqrt(((weighted - neg_ideal) ** 2).sum(axis=1))

    # 相对接近度
    scores = d_neg / (d_pos + d_neg)
    return scores

scores = topsis_evaluate(decision_matrix, weights, criteria_types)

# 排序
ranking = np.argsort(scores)[::-1]
print(f"\nTOPSIS评价结果:")
print(f"{'排名':<6}{'区域':<10}{'得分':<10}")
print("-" * 30)
for rank, idx in enumerate(ranking):
    print(f"  {rank+1:<6}{zone_data.index[idx]:<10}{scores[idx]:.4f}")

# 可视化评价结果
fig, ax = plt.subplots(figsize=(8, 5))
sorted_indices = np.argsort(scores)[::-1]
sorted_zones = [zone_data.index[i] for i in sorted_indices]
sorted_scores = scores[sorted_indices]

bars = ax.barh(range(len(sorted_zones)), sorted_scores,
               color=plt.cm.RdYlGn(sorted_scores / sorted_scores.max()),
               edgecolor='black', linewidth=0.5)
ax.set_yticks(range(len(sorted_zones)))
ax.set_yticklabels(sorted_zones, fontsize=12)
ax.set_xlabel('TOPSIS 相对接近度', fontsize=13)
ax.set_title('各区域出租车运营效率评价', fontsize=14, fontweight='bold')
ax.grid(True, axis='x', alpha=0.3)
ax.invert_yaxis()

for i, (bar, score) in enumerate(zip(bars, sorted_scores)):
    ax.text(score + 0.005, i, f'{score:.4f}', va='center', fontsize=11)

plt.tight_layout()
plt.savefig('topsis_result.png', dpi=300)
plt.show()
print("评价结果图已保存为 topsis_result.png")
```

### 模型二：出租车调度优化（线性规划）

```python
# ====== 模型二：出租车调度优化 ======
print("\n--- 模型二：出租车调度优化（线性规划） ---")

# 问题描述：
# 总共有1000辆出租车需要分配到6个区域
# 目标：最小化总等待时间
# 约束：各区域分配量 >= 最低需求，总分配 = 1000

# 各区域参数
zone_names = zone_data.index.tolist()
demands = zone_data['乘客需求量'].values  # 各区域需求量
min_taxis = (demands * 0.5).astype(int)   # 各区域最低车辆需求（需求的50%）
total_taxis = 1000  # 总可用车辆

print(f"总可用出租车: {total_taxis}辆")
print(f"\n各区域最低需求:")
for name, min_t in zip(zone_names, min_taxis):
    print(f"  {name}: 至少{min_t}辆")

# 目标：最小化总等待时间
# 等待时间与车辆数的关系：wait_time ≈ a / taxi_count
# 简化为：总等待时间 = sum(demand_i / taxi_i * coefficient)
# 即最小化 sum(demand_i / taxi_i)，等价于最小化 sum(demand_i * (1/taxi_i))
# 这是一个非线性问题，但可以简化为线性：

# 简化模型：最小化各区域的"需求/车辆"之和
# 由于1/taxi是非线性的，我们用另一种方式：
# 最小化各区域"超额需求"（需求 - 车辆，取正部分）之和
# 即 min sum(max(0, demand_i - taxi_i))
# 可以引入辅助变量 t_i >= demand_i - taxi_i, t_i >= 0
# min sum(t_i)
# s.t. t_i >= demand_i - taxi_i
#      t_i >= 0
#      sum(taxi_i) = total_taxis
#      taxi_i >= min_taxis_i

from scipy.optimize import linprog

n_zones = len(zone_names)

# 变量：[taxi_0, taxi_1, ..., taxi_5, t_0, t_1, ..., t_5]
# 共 2*n_zones 个变量

# 目标函数：min sum(t_i)
c = np.zeros(2 * n_zones)
c[n_zones:] = 1  # 只最小化 t_i 部分

# 等式约束：sum(taxi_i) = total_taxis
A_eq = np.zeros((1, 2 * n_zones))
A_eq[0, :n_zones] = 1
b_eq = [total_taxis]

# 不等式约束：
# t_i >= demand_i - taxi_i  =>  taxi_i - t_i >= -demand_i  =>  -taxi_i + t_i <= demand_i
# A_ub 形式: A_ub @ x <= b_ub
A_ub = np.zeros((n_zones, 2 * n_zones))
for i in range(n_zones):
    A_ub[i, i] = -1          # -taxy_i
    A_ub[i, n_zones + i] = 1  # +t_i
b_ub = demands

# 变量边界
bounds = []
for i in range(n_zones):
    bounds.append((min_taxis[i], None))  # taxi_i >= min_taxis_i
for i in range(n_zones):
    bounds.append((0, None))              # t_i >= 0

# 求解
result = linprog(c, A_ub=A_ub, b_ub=b_ub, A_eq=A_eq, b_eq=b_eq,
                 bounds=bounds, method='highs')

if result.success:
    taxi_allocation = result.x[:n_zones].astype(int)
    unmet_demand = result.x[n_zones:]

    print(f"\n=== 最优调度方案 ===")
    print(f"{'区域':<10}{'分配车辆':<10}{'需求量':<10}{'未满足需求':<12}{'满足率':<10}")
    print("-" * 55)
    for i in range(n_zones):
        rate = min(100, taxi_allocation[i] / demands[i] * 100)
        print(f"{zone_names[i]:<10}{taxi_allocation[i]:<10}{demands[i]:<10.0f}"
              f"{unmet_demand[i]:<12.1f}{rate:<10.1f}%")
    print("-" * 55)
    print(f"{'合计':<10}{taxi_allocation.sum():<10}{demands.sum():<10.0f}"
          f"{unmet_demand.sum():<12.1f}")
    print(f"\n总未满足需求: {unmet_demand.sum():.1f} 人次")
else:
    print(f"求解失败: {result.message}")

# 可视化调度结果
fig, ax = plt.subplots(figsize=(10, 6))

x = np.arange(n_zones)
width = 0.35

bars1 = ax.bar(x - width/2, demands, width, label='乘客需求量',
               color='#FF7043', edgecolor='black', alpha=0.8)
bars2 = ax.bar(x + width/2, taxi_allocation, width, label='分配出租车数',
               color='#42A5F5', edgecolor='black', alpha=0.8)

ax.set_xticks(x)
ax.set_xticklabels(zone_names, fontsize=11)
ax.set_ylabel('数量', fontsize=13)
ax.set_title('出租车最优调度方案', fontsize=14, fontweight='bold')
ax.legend(fontsize=12)
ax.grid(True, axis='y', alpha=0.3)

# 标注数值
for bar in bars1:
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 5,
            f'{bar.get_height():.0f}', ha='center', fontsize=9)
for bar in bars2:
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 5,
            f'{bar.get_height():.0f}', ha='center', fontsize=9)

plt.tight_layout()
plt.savefig('optimal_allocation.png', dpi=300)
plt.show()
print("调度方案图已保存为 optimal_allocation.png")
```

```python
# ====== 模型三：区域聚类分析 ======
print("\n--- 模型三：区域聚类分析 ---")

# 将各区域按运营特征聚类
```

## 小结

本节用一个完整的"出租车调度优化"案例演示了编程成员的全流程工作：

1. **数据生成与预处理**：生成模拟数据，清洗、添加派生特征
2. **探索性数据分析**：画统计图表，发现数据规律
3. **模型实现**：TOPSIS评价 + 线性规划优化 + KMeans聚类
4. **结果可视化**：综合图表展示分析结果
5. **结果导出**：保存为CSV/Excel/图片供论文使用

> **避坑指南**：比赛中最重要的是**时间管理**。不要在某个步骤上卡太久——如果某个模型跑不通，先换一个简单的模型跑出结果，再回来优化。先有结果，再优化，不要追求完美。
