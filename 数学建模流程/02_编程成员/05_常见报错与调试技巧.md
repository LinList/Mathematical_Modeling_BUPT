## 一、常见报错及解决方案

### 1.1 维度不匹配（ValueError: shapes not aligned）

这是数学建模中最常见的错误，出现在矩阵运算中。

```python
import numpy as np

# ====== 错误示例 ======
A = np.array([[1, 2, 3],
              [4, 5, 6]])      # 形状 (2, 3)

B = np.array([[1, 2],
              [3, 4],
              [5, 6],
              [7, 8]])          # 形状 (4, 2)

# C = A @ B  # 报错！ValueError: matmul: Input operand 1 has a mismatch
#              in its core dimension 0, with gusr signature (n,k),(k,m)->(n,m)
#              (size 3 is different from 4)

# ====== 解决方案 ======
# 矩阵乘法要求：A的列数 = B的行数
# A 是 (2,3)，B 需要是 (3, n) 才能相乘

# 检查维度
print(f"A 形状: {A.shape}")  # (2, 3)
print(f"B 形状: {B.shape}")  # (4, 2)

# 方法1：修改B使其维度匹配
B_correct = B[:3, :]  # 取前3行，形状变为 (3, 2)
C = A @ B_correct      # (2, 3) @ (3, 2) = (2, 2) ✓
print(f"结果形状: {C.shape}")

# 方法2：使用转置
# A @ B.T → (2,3) @ (2,4) → 还是不行
# A.T @ B → (3,2) @ (4,2) → 还是不行
# 需要根据实际数学公式来调整

# ====== 常见场景：权重矩阵与数据矩阵相乘 ======
# 数据矩阵 data: (n_samples, n_features)
# 权重向量 weights: (n_features,)
# 加权得分 = data @ weights → (n_samples,)

data = np.random.rand(100, 5)      # 100个样本，5个特征
weights = np.random.rand(5)         # 5个权重

# 正确写法
scores = data @ weights              # 形状 (100,)
print(f"加权得分形状: {scores.shape}")

# 错误写法（维度不匹配）
# scores = data @ weights.reshape(-1, 1)  # (100,5) @ (5,1) = (100,1) ← 这个其实也行
```

> **排错指南**：遇到维度不匹配错误时，第一时间 `print(arr.shape)` 检查每个数组的形状。记住矩阵乘法规则：`(m,n) @ (n,p) = (m,p)`，中间维度必须一致。

### 1.2 索引越界（IndexError）

```python
import numpy as np

# ====== 错误示例 ======
arr = np.array([10, 20, 30, 40, 50])  # 长度5，索引0~4

# print(arr[5])  # IndexError: index 5 is out of bounds for axis 0 with size 5

# ====== 原因分析 ======
# Python索引从0开始，长度为5的数组，索引范围是0~4
# arr[5] 超出了范围

# ====== 解决方案 ======
# 方法1：检查长度
print(f"数组长度: {len(arr)}")
print(f"最后一个元素: {arr[len(arr) - 1]}")  # arr[4]

# 方法2：用负索引从末尾访问
print(f"最后一个元素: {arr[-1]}")   # 50
print(f"倒数第二个: {arr[-2]}")     # 40

# 方法3：切片时不会越界
print(f"取前10个（自动截断）: {arr[:10]}")  # 不会报错，返回全部5个

# ====== DataFrame 索引越界 ======
import pandas as pd

df = pd.DataFrame({'A': [1, 2, 3], 'B': [4, 5, 6]})
print(f"\nDataFrame 行数: {len(df)}")

# 错误：loc 用标签索引
# df.loc[5]  # KeyError: 5

# 正确：loc 用标签（默认是0,1,2...）
print(df.loc[2])  # 第3行

# 错误：iloc 用位置索引
# df.iloc[3]  # IndexError: single positional indexer is out-of-bounds

# 正确
print(df.iloc[2])  # 第3行
```

> **排错指南**：`loc` 用标签索引（行名/列名），`iloc` 用位置索引（整数）。如果不确定有多少行，先用 `len(df)` 或 `df.shape` 检查。

### 1.3 NaN 值导致的计算错误

```python
import numpy as np
import pandas as pd

# ====== 错误示例 ======
data = np.array([1, 2, np.nan, 4, 5])

# NaN会"污染"整个计算
print(f"求和: {np.sum(data)}")       # nan
print(f"均值: {np.mean(data)}")      # nan
print(f"最大值: {np.max(data)}")     # nan

# ====== 解决方案 ======

# 方法1：用 nan-safe 函数
print(f"\nnan-safe 求和: {np.nansum(data)}")    # 12.0
print(f"nan-safe 均值: {np.nanmean(data)}")     # 3.0
print(f"nan-safe 最大值: {np.nanmax(data)}")    # 5.0

# 方法2：先删除NaN再计算
data_clean = data[~np.isnan(data)]
print(f"\n删除NaN后: {data_clean}")
print(f"求和: {data_clean.sum()}")

# 方法3：填充NaN
data_filled = np.nan_to_num(data, nan=0)
print(f"\n填充0后: {data_filled}")

# 方法4：pandas自动处理
s = pd.Series([1, 2, np.nan, 4, 5])
print(f"\npandas求和（自动跳过NaN）: {s.sum()}")  # 12.0
print(f"pandas均值（自动跳过NaN）: {s.mean()}")   # 3.0

# ====== 检查是否有NaN ======
print(f"\n是否有NaN: {np.isnan(data).any()}")     # True
print(f"NaN个数: {np.isnan(data).sum()}")         # 1
```

> **避坑指南**：计算结果如果是 `nan`，第一时间检查输入数据是否有缺失值。`np.sum()` 遇到 NaN 会返回 nan，但 `pd.Series.sum()` 会自动跳过 NaN。这是 NumPy 和 Pandas 的重要区别。

### 1.5 数据类型问题

```python
import pandas as pd
import numpy as np

# ====== 问题1：字符串被当作数值 ======
df = pd.DataFrame({
    '数值列': ['10', '20', '30', '40'],   # 字符串类型
    '混合列': ['1', '2', 'abc', '4'],     # 混合类型
})

print("原始数据类型:")
print(df.dtypes)  # 都是 object（字符串）

# 错误：直接做数学运算
# df['数值列'] + 100  # 得到 "101002003040"（字符串拼接！）

# 解决方案：转换数据类型
df['数值列'] = df['数值列'].astype(int)
print(f"\n转换后: {df['数值列'].dtype}")
print(f"加法结果: {(df['数值列'] + 100).tolist()}")

# 混合列转换（含非数字字符串）
# df['混合列'].astype(int)  # ValueError: invalid literal for int()
# 解决方案：用 to_numeric，设置 errors='coerce' 将无效值转为NaN
df['混合列'] = pd.to_numeric(df['混合列'], errors='coerce')
print(f"\n混合列转换后: {df['混合列'].tolist()}")  # [1.0, 2.0, nan, 4.0]

# ====== 问题2：浮点数精度 ======
print(f"\n=== 浮点数精度问题 ===")
print(f"0.1 + 0.2 = {0.1 + 0.2}")           # 0.30000000000000004
print(f"0.1 + 0.2 == 0.3: {0.1 + 0.2 == 0.3}")  # False!

# 解决方案：用 np.isclose 比较浮点数
print(f"np.isclose(0.1+0.2, 0.3): {np.isclose(0.1 + 0.2, 0.3)}")  # True

# 或用 round 取整
print(f"round(0.1+0.2, 2) == 0.3: {round(0.1 + 0.2, 2) == 0.3}")  # True

# ====== 问题3：整数除法 ======
print(f"\n=== 整数除法问题 ===")
print(f"3 / 2 = {3 / 2}")      # 1.5（Python3 默认浮点除法）
print(f"3 // 2 = {3 // 2}")    # 1（整数除法，向下取整）
print(f"-3 // 2 = {-3 // 2}")  # -2（注意：向下取整，不是向零取整！）

# ====== 问题4：DataFrame 修改视图 vs 副本 ======
df2 = pd.DataFrame({'A': [1, 2, 3], 'B': [4, 5, 6]})

# 警告：SettingWithCopyWarning
# df2[df2['A'] > 1]['B'] = 100  # 可能不生效，会报警告

# 正确写法：用 loc
df2.loc[df2['A'] > 1, 'B'] = 100
print(f"\n正确修改后:\n{df2}")
```

> **排错指南**：拿到数据后，第一时间用 `df.dtypes` 检查每列的数据类型。如果数值列显示为 `object`，说明被读成了字符串，需要用 `astype()` 转换。

### 2.1 print 大法（最简单最有效）

```python
import numpy as np

# ====== 示例：调试一个计算过程 ======

def calculate_weighted_score(data, weights):
    """计算加权得分（有bug的版本）"""
    print(f"[DEBUG] 输入数据形状: {data.shape}")
    print(f"[DEBUG] 权重形状: {weights.shape}")
    print(f"[DEBUG] 权重值: {weights}")
    print(f"[DEBUG] 权重和: {weights.sum()}")

    # 标准化数据
    data_normalized = (data - data.min(axis=0)) / (data.max(axis=0) - data.min(axis=0))
    print(f"[DEBUG] 标准化后数据范围: [{data_normalized.min():.3f}, {data_normalized.max():.3f}]")

    # 加权
    weighted = data_normalized * weights
    print(f"[DEBUG] 加权后数据形状: {weighted.shape}")

    # 求和得到得分
    scores = weighted.sum(axis=1)
    print(f"[DEBUG] 得分形状: {scores.shape}")
    print(f"[DEBUG] 得分范围: [{scores.min():.3f}, {scores.max():.3f}]")

    return scores


# 测试
np.random.seed(42)
test_data = np.random.rand(5, 3) * 100
test_weights = np.array([0.5, 0.3, 0.2])

result = calculate_weighted_score(test_data, test_weights)
print(f"\n最终结果: {result}")
```

> **小贴士**：`print` 调试的核心思路是"追踪数据流"——在每一步操作后打印数据的形状、范围、类型，看哪一步出了问题。比任何高级调试工具都简单有效。

### 2.3 try-except 异常处理

```python
import numpy as np
import pandas as pd

# ====== try-except 基本用法 ======

def safe_divide(a, b):
    """安全除法，避免除以零"""
    try:
        return a / b
    except ZeroDivisionError:
        print(f"警告: 除以零！a={a}, b={b}，返回0")
        return 0
    except TypeError as e:
        print(f"类型错误: {e}")
        return None

print(safe_divide(10, 2))   # 5.0
print(safe_divide(10, 0))   # 警告 + 返回0

# ====== 实用场景：安全读取文件 ======
def safe_read_csv(filepath):
    """安全读取CSV文件"""
    try:
        df = pd.read_csv(filepath, encoding='utf-8')
        print(f"成功读取（utf-8）: {filepath}")
        return df
    except UnicodeDecodeError:
        try:
            df = pd.read_csv(filepath, encoding='gbk')
            print(f"成功读取（gbk）: {filepath}")
            return df
        except Exception as e:
            print(f"读取失败: {e}")
            return None
    except FileNotFoundError:
        print(f"文件不存在: {filepath}")
        return None
    except Exception as e:
        print(f"未知错误: {type(e).__name__}: {e}")
        return None

# ====== 实用场景：安全计算 ======
def safe_matrix_multiply(A, B):
    """安全矩阵乘法"""
    try:
        result = A @ B
        return result
    except ValueError as e:
        print(f"维度不匹配: A{A.shape} @ B{B.shape}")
        print(f"错误: {e}")
        # 尝试自动修复：转置B
        if A.shape[1] == B.shape[0]:
            return A @ B
        elif A.shape[1] == B.shape[1]:
            print("尝试转置B...")
            return A @ B.T
        else:
            print("无法自动修复，请检查矩阵维度")
            return None

# 测试
A = np.random.rand(3, 4)
B = np.random.rand(4, 2)
result = safe_matrix_multiply(A, B)
print(f"结果形状: {result.shape if result is not None else 'None'}")

B_wrong = np.random.rand(2, 4)
result2 = safe_matrix_multiply(A, B_wrong)
```

> **小贴士**：在比赛中，用 `try-except` 包裹可能出错的操作（文件读取、矩阵运算、类型转换），可以防止程序崩溃。但不要滥用——只包裹"可能出错且你知道怎么处理"的代码。

### 2.4 调试检查清单

```python
# ====== 调试时的检查清单 ======

def debug_checklist(data):
    """数据调试检查清单"""
    print("=== 数据调试检查清单 ===\n")

    # 1. 检查数据类型
    print(f"1. 数据类型: {type(data)}")

    # 2. 检查形状
    if hasattr(data, 'shape'):
        print(f"2. 数据形状: {data.shape}")
    elif hasattr(data, '__len__'):
        print(f"2. 数据长度: {len(data)}")

    # 3. 检查前几行
    print(f"3. 前5个值:\n{data[:5] if hasattr(data, '__getitem__') else data}")

    # 4. 检查缺失值
    if isinstance(data, pd.DataFrame):
        print(f"4. 缺失值:\n{data.isnull().sum()}")
    elif isinstance(data, np.ndarray):
        print(f"4. NaN数量: {np.isnan(data).sum()}")

    # 5. 检查数值范围
    if isinstance(data, np.ndarray) and data.dtype in [np.float64, np.int64, float, int]:
        print(f"5. 数值范围: [{data.min():.4f}, {data.max():.4f}]")
        print(f"   均值: {data.mean():.4f}, 标准差: {data.std():.4f}")

    # 6. 检查是否有 inf
    if isinstance(data, np.ndarray):
        inf_count = np.isinf(data).sum()
        if inf_count > 0:
            print(f"6. ⚠ 发现 {inf_count} 个无穷大值！")
        else:
            print(f"6. 无无穷大值 ✓")
```

## 三、性能优化

### 3.1 向量化运算 vs 循环

```python
import numpy as np
import time

# ====== 对比：循环 vs 向量化 ======

# 任务：计算100万个数的平方

# 方法1：用 for 循环（慢）
start = time.time()
result_loop = np.zeros(1000000)
for i in range(1000000):
    result_loop[i] = i ** 2
time_loop = time.time() - start
print(f"for循环耗时: {time_loop:.4f}秒")

# 方法2：用列表推导式（较快）
start = time.time()
result_list = [i ** 2 for i in range(1000000)]
time_list = time.time() - start
print(f"列表推导式耗时: {time_list:.4f}秒")

# 方法3：用 NumPy 向量化（最快）
start = time.time()
arr = np.arange(1000000)
result_vec = arr ** 2
time_vec = time.time() - start
print(f"NumPy向量化耗时: {time_vec:.4f}秒")

print(f"\n向量化比循环快 {time_loop / time_vec:.0f} 倍！")

# ====== 实际场景：计算距离矩阵 ======
print("\n=== 距离矩阵计算 ===")

# 1000个点的两两距离
n = 1000
points = np.random.rand(n, 2)

# 方法1：双重循环（慢）
start = time.time()
dist_loop = np.zeros((n, n))
for i in range(n):
    for j in range(n):
        dist_loop[i, j] = np.sqrt((points[i, 0] - points[j, 0])**2 +
                                   (points[i, 1] - points[j, 1])**2)
time_loop = time.time() - start
print(f"循环方式: {time_loop:.4f}秒")

# 方法2：向量化（快）
start = time.time()
# 利用广播机制：(n,1,2) - (1,n,2) = (n,n,2)
diff = points[:, np.newaxis, :] - points[np.newaxis, :, :]
dist_vec = np.sqrt(np.sum(diff ** 2, axis=2))
time_vec = time.time() - start
print(f"向量化方式: {time_vec:.4f}秒")
print(f"加速比: {time_loop / time_vec:.0f}倍")

# 方法3：用 scipy（更快）
from scipy.spatial.distance import cdist
start = time.time()
dist_scipy = cdist(points, points)
time_scipy = time.time() - start
print(f"scipy方式: {time_scipy:.4f}秒")
```

> **小贴士**：NumPy 向量化运算通常比 Python 循环快 10~100 倍。能用向量运算就不要用循环！但如果数据量很小（<1000），差别不大，怎么方便怎么来。

## 五、代码模板：完整的健壮数据处理流程

```python
import numpy as np
import pandas as pd
import warnings
import traceback

# 屏蔽不影响结果的警告
warnings.filterwarnings('ignore', category=FutureWarning)

def robust_data_pipeline(filepath=None, df=None):
    """
    健壮的数据处理流程模板
    包含：读取 → 检查 → 清洗 → 验证
    """
    print("=" * 50)
    print("开始数据处理流程")
    print("=" * 50)

    try:
        # ====== 第一步：读取数据 ======
        if df is None and filepath:
            print(f"\n[1/5] 读取文件: {filepath}")
            try:
                df = pd.read_csv(filepath, encoding='utf-8')
            except UnicodeDecodeError:
                df = pd.read_csv(filepath, encoding='gbk')
            except Exception as e:
                print(f"读取失败: {e}")
                return None

        print(f"数据形状: {df.shape}")
        print(f"列名: {df.columns.tolist()}")

        # ====== 第二步：检查数据 ======
        print(f"\n[2/5] 数据检查")
        print(f"数据类型:\n{df.dtypes}")
        print(f"\n缺失值:\n{df.isnull().sum()}")
        print(f"\n重复行: {df.duplicated().sum()}")

        # ====== 第三步：数据清洗 ======
        print(f"\n[3/5] 数据清洗")

        # 删除重复行
        if df.duplicated().sum() > 0:
            df = df.drop_duplicates()
            print(f"  已删除重复行")

        # 处理缺失值
        numeric_cols = df.select_dtypes(include=[np.number]).columns
        for col in numeric_cols:
            if df[col].isnull().sum() > 0:
                median_val = df[col].median()
                df[col] = df[col].fillna(median_val)
                print(f"  {col}: 用中位数{median_val:.2f}填充缺失值")

        # 处理无穷大
        for col in numeric_cols:
            if np.isinf(df[col]).any():
                df[col] = df[col].replace([np.inf, -np.inf], df[col].median())
                print(f"  {col}: 替换无穷大值")

        # ====== 第四步：数据验证 ======
        print(f"\n[4/5] 数据验证")
        print(f"  清洗后形状: {df.shape}")
        print(f"  缺失值总数: {df.isnull().sum().sum()}")
        print(f"  数值列统计:\n{df[numeric_cols].describe().round(2)}")

        # ====== 第五步：完成 ======
        print(f"\n[5/5] 数据处理完成！")
        print(f"最终数据形状: {df.shape}")

        return df

    except Exception as e:
        print(f"\n✗ 处理过程中出错:")
        print(f"  错误类型: {type(e).__name__}")
        print(f"  错误信息: {e}")
        print(f"\n详细堆栈:")
        traceback.print_exc()
        return None


# 使用示例
if __name__ == "__main__":
    # 创建测试数据
    np.random.seed(42)
    test_df = pd.DataFrame({

    })

    # 故意加入一些问题数据

    test_df.loc[1, 'B'] = np.inf
    test_df.loc[2, 'A'] = np.nan
    test_df.duplicated()  # 可能有一些重复

    # 运行处理流程
    result = robust_data_pipeline(df=test_df)

    if result is not None:
        print(f"\n处理成功！最终数据:")
        print(result.head())
    else:
        print("\n处理失败！")
```
